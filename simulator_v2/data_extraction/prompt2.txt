//  Przykład jak wykorzystać wylosowane dane do generowania pytań (do użycia z modelem AI)
function generujPytania(wynikiLosowan) {
  let pytania = [];
  for (const lp in wynikiLosowan) {
    if (wynikiLosowan.hasOwnProperty(lp)) {
      const wynik = wynikiLosowan[lp];
      const pytanie = `Jak oceniłbyś/oceniłabyś zdolność pacjenta do ${wynik.activity.toLowerCase()}? Pacjent ${wynik.description.toLowerCase()}? (Wynik: ${wynik.score})`;
      pytania.push(pytanie);
    }
  }
  return pytania;
}
